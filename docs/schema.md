# Database Schema (DuckDB)

## `raw_prices` — Stock prices and volume
```sql
CREATE TABLE raw_prices (
    ticker       VARCHAR,   -- e.g. '7203.T', 'XOM'
    market       VARCHAR,   -- 'JP' or 'US'
    date         DATE,
    open         DOUBLE,
    high         DOUBLE,
    low          DOUBLE,
    close        DOUBLE,
    volume       BIGINT,
    PRIMARY KEY (ticker, date)
);
```

## `raw_oil_prices` — Crude oil prices
```sql
CREATE TABLE raw_oil_prices (
    date         DATE,
    wti          DOUBLE,    -- WTI crude (CL=F)
    brent        DOUBLE,    -- Brent crude (BZ=F)
    PRIMARY KEY (date)
);
```

## `sector_indices` — Sector-level indices (1970s~)
```sql
CREATE TABLE sector_indices (
    date         DATE,
    market       VARCHAR,   -- 'JP' or 'US'
    sector       VARCHAR,   -- e.g. 'energy', 'auto', 'airline'
    index_value  DOUBLE,
    source       VARCHAR,   -- 'BOJ', 'JPX', 'FRED', 'Stooq'
    PRIMARY KEY (market, sector, date)
);
```

## `universe` — Target stock list
```sql
CREATE TABLE universe (
    ticker       VARCHAR,
    market       VARCHAR,   -- 'JP' or 'US'
    name         VARCHAR,
    sector       VARCHAR,
    index_name   VARCHAR,   -- 'Nikkei225', 'TOPIX500', 'SP500'
    PRIMARY KEY (ticker)
);
```

## `indicators` — Technical indicators (computed by Polars, written to DuckDB)
```sql
CREATE TABLE indicators (
    ticker       VARCHAR,
    date         DATE,
    ma5          DOUBLE,
    ma20         DOUBLE,
    ma50         DOUBLE,
    ma200        DOUBLE,
    rsi14        DOUBLE,
    macd         DOUBLE,
    macd_signal  DOUBLE,
    bb_upper     DOUBLE,
    bb_lower     DOUBLE,
    volume_ratio DOUBLE,    -- volume / 20-day avg volume
    PRIMARY KEY (ticker, date)
);
```

## `sector_sensitivity` — Oil price sensitivity per sector (output of Phase 2)
```sql
CREATE TABLE sector_sensitivity (
    market       VARCHAR,
    sector       VARCHAR,
    event        VARCHAR,   -- 'oil_shock_1973', 'oil_shock_1979'
    correlation  DOUBLE,    -- correlation coefficient with oil price
    beta         DOUBLE,    -- beta relative to oil price
    PRIMARY KEY (market, sector, event)
);
```

## `scores` — Stock scores (generated by dbt)
```sql
CREATE TABLE scores (
    ticker           VARCHAR,
    date             DATE,
    technical_score  DOUBLE,
    sentiment_score  DOUBLE,
    final_score      DOUBLE,
    signal           VARCHAR,   -- 'BUY', 'SELL', 'HOLD'
    PRIMARY KEY (ticker, date)
);
```

## `news_events` — News and sentiment
```sql
CREATE TABLE news_events (
    id           VARCHAR,
    published    TIMESTAMP,
    source       VARCHAR,   -- 'Reuters', 'AP'
    title        VARCHAR,
    sentiment    DOUBLE,    -- VADER score (-1.0 to 1.0)
    keywords     VARCHAR[], -- matched keywords
    PRIMARY KEY (id)
);
```
